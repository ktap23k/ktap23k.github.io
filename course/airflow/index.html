# ⚠️ Anti-pattern: KHÔNG import heavy libs ở top-level
def validate_invoice_batch(**context):

1.4 TaskFlow API (Modern approach)

# dags/stage1_taskflow.py
from airflow.sdk import dag, task
from datetime import datetime, timedelta

@dag(
    schedule=timedelta(hours=1),
    start_date=datetime(2026, 6, 1),
    catchup=False,
    tags=['stage1', 'taskflow'],
)
def stage1_taskflow_etl():

    @task
    def extract() -> dict:
        return {"records": 1000, "source": "kafka-topic-invoices"}

    @task
    def transform(data: dict) -> dict:
        return {**data, "processed": True, "valid_count": 950}

    @task
    def load(data: dict) -> None:
        print(f"Loading {data['valid_count']} records to Cassandra")

    load(transform(extract()))

1.5 Key Concepts Checklist

  • DAG là đồ thị có hướng, không chu trình (DAG = Directed Acyclic Graph)

Stage 2: Architecture Deep Dive (Ngày 3-5)

2.1 Scheduler Loop (The Brain)

Scheduler chạy vòng lặp vô hạn:

1. Parse DAG files trong dags_folder
2. Tạo DagRun nếu đến schedule time
3. Tạo TaskInstance cho DagRun
4. Kiểm tra dependencies (upstream success, sensors, trigger rules)
5. Đẩy ready TaskInstance → Executor queue
6. Theo dõi state từ Executor → cập nhật DB
7. Kiểm tra DagRun hoàn thành / thất bại

Cấu hình tối ưu cho scheduler:

# airflow.cfg hoặc env vars
AIRFLOW__SCHEDULER__DAG_DIR_LIST_INTERVAL = 120
AIRFLOW__SCHEDULER__MIN_FILE_PROCESS_INTERVAL = 30
AIRFLOW__SCHEDULER__MAX_DAGRUNS_PER_LOOP_TO_SCHEDULE = 20
AIRFLOW__SCHEDULER__PARSING_PROCESSES = 2
AIRFLOW__SCHEDULER__SCHEDULER_HEARTBEAT_SEC = 5

Pattern A: Airflow orchestrate Kafka → Spark (batch)

# dags/stage3_kafka_spark.py
from airflow.sdk import DAG
from airflow.providers.apache.spark.operators.spark_submit import SparkSubmitOperator
from airflow.providers.standard.operators.python import PythonOperator
from datetime import datetime, timedelta

with DAG(
    'kafka_to_spark_invoice_etl',
    schedule=timedelta(hours=1),
    start_date=datetime(2026, 6, 1),
    catchup=False,
    tags=['stage3', 'kafka', 'spark'],
) as dag:

    def check_kafka_lag(**context):
        import json
        # Trong production: dùng KafkaAdminClient hoặc query monitoring DB
        lag_info = {"topic": "invoices", "lag": 1500, "threshold": 1000}
        if lag_info['lag'] < lag_info['threshold']:
            raise Exception("Not enough new messages, skipping")
        context['ti'].xcom_push(key='lag_info', value=json.dumps(lag_info))
        return lag_info

    check_lag = PythonOperator(
        task_id='check_kafka_lag',
        python_callable=check_kafka_lag,
    )

    spark_job = SparkSubmitOperator(
        task_id='spark_invoice_transform',
        application='/opt/spark/jobs/invoice_etl.py',
        conn_id='spark_default',
        conf={
            'spark.executor.memory': '2g',
            'spark.executor.cores': '2',
        },
        application_args=[
            '--date', '{{ ds }}',
            '--topic', 'invoices',
            '--output', 's3a://processed/invoices/{{ ds }}'
        ],
    )

    def verify_cassandra_load(**context):
        from cassandra.cluster import Cluster
        cluster = Cluster(['cassandra-node1', 'cassandra-node2'])
        session = cluster.connect('invoice_db')
        result = session.execute(
            "SELECT COUNT(*) FROM invoices WHERE date = %s", 
            [context['ds']]
        )
        count = result.one()[0]
        print(f"Loaded {count} records to Cassandra for {context['ds']}")
        assert count > 0, "No records found!"

    verify = PythonOperator(
        task_id='verify_cassandra',
        python_callable=verify_cassandra_load,
    )

    check_lag >> spark_job >> verify

3.3 Custom Operator cho E-invoice XML Validation

# plugins/operators/xml_validator_operator.py
from airflow.models.baseoperator import BaseOperator
from airflow.exceptions import AirflowException
import xml.etree.ElementTree as ET
from pathlib import Path

class XMLValidationOperator(BaseOperator):
    template_fields = ('xml_path', 'schema_path')

    def __init__(self, xml_path: str, schema_path: str, **kwargs):
        super().__init__(**kwargs)
        self.xml_path = xml_path
        self.schema_path = schema_path

    def execute(self, context):
        self.log.info(f"Validating {self.xml_path} against {self.schema_path}")

        xml_file = Path(self.xml_path)
        if not xml_file.exists():
            raise AirflowException(f"XML file not found: {self.xml_path}")

        try:
            tree = ET.parse(xml_file)
            root = tree.getroot()

            # Validate required tags (KHMSHDon, KHHDon, SHDon, NBan/MST)
            required_tags = ['KHMSHDon', 'KHHDon', 'SHDon', 'NBan']
            for tag in required_tags:
                elem = root.find(f'.//{tag}')
                if elem is None or not elem.text:
                    raise AirflowException(f"Missing or empty tag: {tag}")

            # Validate MST (Tax code) format
            mst = root.find('.//NBan/MST')
            if mst is None or len(mst.text) < 10:
                raise AirflowException("Invalid MST format")

            self.log.info("XML validation passed")
            return {"file": self.xml_path, "valid": True, "mst": mst.text}

        except ET.ParseError as e:
            raise AirflowException(f"XML parse error: {e}")

3.4 HTTP/REST API Integration (mTLS + httpx)

# dags/stage3_api_integration.py
from airflow.sdk import DAG
from airflow.providers.standard.operators.python import PythonOperator
from datetime import datetime, timedelta
import httpx
import ssl

with DAG(
    'api_invoice_sync',
    schedule=timedelta(minutes=30),
    start_date=datetime(2026, 6, 1),
    catchup=False,
    tags=['stage3', 'api', 'mtls'],
) as dag:

    def sync_invoices(**context):
        # Sử dụng httpx với mTLS — phù hợp với hệ thống của bạn
        cert_path = '/certs/client.crt'
        key_path = '/certs/client.key'
        ca_path = '/certs/ca.crt'

        context_ssl = ssl.create_default_context(cafile=ca_path)
        context_ssl.load_cert_chain(cert_path, key_path)

        with httpx.Client(
            http2=True,  # HTTP/2 như hệ thống của bạn
            verify=context_ssl,
            timeout=30.0
        ) as client:
            response = client.post(
                'https://api.einvoice.gov.vn/sync',
                json={"date": context['ds'], "batch_size": 1000}
            )
            response.raise_for_status()
            return response.json()

    sync_task = PythonOperator(
        task_id='sync_via_mtls',
        python_callable=sync_invoices,
    )

4.5 SLA & Alerting (Airflow 3.2 — DeadlineAlert)

Airflow 3.2 thay thế cơ chế SLA cũ bằng DeadlineAlert — cho phép định nghĩa deadline linh hoạt và callback đồng bộ/bất đồng bộ.

# dags/stage4_deadline_alert.py
from airflow.sdk import DAG
from airflow.providers.standard.operators.python import PythonOperator
from airflow.sdk.definitions.deadline import DeadlineAlert, DeadlineReference, SyncCallback
from datetime import datetime, timedelta


def on_deadline_miss(*, deadline_reference: DeadlineReference, **context):
    """Callback được gọi khi task vượt deadline."""
    print(f"⚠️ Deadline missed: {deadline_reference.name}")
    # Gửi alert: PagerDuty, Slack, Opsgenie...


with DAG(
    dag_id='production_invoice_pipeline',
    schedule=timedelta(hours=1),
    start_date=datetime(2026, 6, 1),
    catchup=False,
    tags=['production', 'invoice'],
    default_args={
        'owner': 'tuanta',
        'retries': 2,
        'retry_delay': timedelta(minutes=5),
    },
) as dag:

    # Deadline cấp DAG: toàn bộ DAG phải xong trong 45 phút
    dag.deadline = DeadlineAlert(
        deadline=timedelta(minutes=45),
        name='dag_finish_deadline',
        callback=SyncCallback(on_deadline_miss),
    )

    def calculate_tax(**context):
        import time
        time.sleep(5)
        return {"tax": 1000000}

    # Task-critical với deadline riêng
    critical_task = PythonOperator(
        task_id='tax_calculation',
        python_callable=calculate_tax,
        deadline=DeadlineAlert(
            deadline=timedelta(minutes=15),
            name='tax_calc_deadline',
            callback=SyncCallback(on_deadline_miss),
        ),
    )

Lưu ý:

  • DeadlineAlert hỗ trợ cả SyncCallback và AsyncCallback (cho I/O không chặn).
  • Thứ tự ưu tiên: task deadline > DAG deadline.
  • Kết hợp với email/Slack webhook trong callback để cảnh báo tức thì.

from airflow.sdk import Asset
from airflow.sdk import dag, task

# Define datasets
invoice_raw = Asset("s3://bucket/invoices/raw/{{ ds }}")

    @task(outlets=[invoice_processed])
    def transform():
        print("Processing because raw dataset updated")
        return "done"

    transform()

5.2 Dynamic Task Mapping

from airflow.sdk import dag, task
from datetime import datetime

@dag(
    schedule='@daily',
    start_date=datetime(2026, 6, 1),
    catchup=False,
    tags=['stage5', 'dynamic'],
)
def dynamic_invoice_processing():

    @task
    def get_files_to_process(ds: str) -> list:
        # Trong production: list S3 bucket hoặc query DB
        return [
            f"s3://bucket/invoices/{ds}/file_{i}.xml" 
            for i in range(1, 6)
        ]

    @task
    def process_file(file_path: str) -> dict:
        # Mỗi file = 1 task instance
        print(f"Processing {file_path}")
        return {"file": file_path, "status": "ok"}

    @task
    def aggregate(results: list) -> dict:
        return {"total": len(results), "results": results}

    files = get_files_to_process("{{ ds }}")
    processed = process_file.expand(file_path=files)  # Dynamic mapping!
    aggregate(processed)

dynamic_invoice_processing()

5.4 RAG / AI Pipeline Orchestration

from airflow.sdk import dag, task
from datetime import datetime

@dag(
    schedule='@daily',
    start_date=datetime(2026, 6, 1),
    catchup=False,
    tags=['stage5', 'ai', 'rag'],
)
def rag_document_pipeline():

    @task
    def extract_docs() -> list:
        return ["doc1.pdf", "doc2.pdf", "doc3.pdf"]

    @task
    def chunk_and_embed(doc: str) -> dict:
        # Gọi local LLM (llama.cpp/Qwen) hoặc API
        return {"doc": doc, "chunks": 10, "vectors": "..."}

    @task
    def index_to_vector_db(data: dict) -> None:
        print(f"Indexing {data['doc']} with {data['chunks']} chunks")

    docs = extract_docs()
    embedded = chunk_and_embed.expand(doc=docs)
    index_to_vector_db.expand(data=embedded)

6.2 When to Choose What

Chọn Airflow khi:

  • Team > 5 người, cần enterprise governance
  • Existing infrastructure: Kafka, Redis, K8s
  • Complex cross-team DAG dependencies
  • Cần ecosystem rộng (200+ providers)
  • Batch ETL là core workflow

Chọn Prefect khi:

  • Team nhỏ (2-8), cần nhanh gọn
  • ML/AI pipelines, dynamic workflows
  • Cost-sensitive (chạy local/cloud hybrid)
  • Không muốn manage scheduler/worker infra

Chọn Dagster khi:

  • Data platform greenfield
  • Cần data lineage & quality testing nghiêm ngặt
  • Data mesh architecture
  • Asset-centric thinking (data as product)

Appendix: Anti-patterns & Troubleshooting

A.1 Top 10 Anti-patterns

  1. Import pandas/spark ở top-level DAG → Làm scheduler parse chậm. Fix: lazy import trong task.
  2. XCom chứa data lớn → DB bloat. Fix: Lưu S3/Redis, XCom chỉ lưu URI.
  3. Catchup=True mặc định → Chạy lại hàng trăm historical runs. Fix: catchup=False.
  4. Sensor poke mode không timeout → Chiếm worker slot mãi. Fix: mode='reschedule' hoặc deferrable=True.
  5. Chạy heavy computation trong worker → Kill worker. Fix: Submit Spark/K8s job.
  6. Hardcode credentials → Security risk. Fix: Airflow Connections + Vault.
  7. Không đặt retries → Task fail 1 lần = pipeline fail. Fix: retries=1-3.
  8. Dùng Variable trong top-level DAG → Query DB mỗi lần parse. Fix: Dùng Jinja template {{ var.value.x }}.
  9. DAG quá lớn → > 100 tasks = khó debug. Fix: Split thành sub-DAGs hoặc TaskGroup.
  10. Không monitor scheduler lag → DAG chạy trễ không biết. Fix: Prometheus + alert.

A.3 Performance Tuning Checklist

  • AIRFLOW__SCHEDULER__PARSING_PROCESSES = số CPU core / 2
  • AIRFLOW__SCHEDULER__DAG_DIR_LIST_INTERVAL >= 120s nếu DAG ít thay đổi
  • AIRFLOW__CORE__PARALLELISM <= tổng worker slots
  • AIRFLOW__DATABASE__SQL_ALCHEMY_POOL_SIZE >= 10
  • Dùng DAG serialization để giảm parse time
  • Enable schedule caching cho DAGs ổn định
  • Dùng deferrable sensors thay vì poke
  • XCom backend = S3/Redis cho data > 1KB
  • Task duration < 1 hour (nếu dài hơn, cân nhắc split)
  • Log rotation: AIRFLOW__LOGGING__REMOTE_LOGGING = True → S3/CloudWatch
